src/main/java/com/zoo/zims/
├── common/                 # [公共模块]
│   ├── Result.java         # 统一响应结果封装类 (JSON格式: code, msg, data)
│   ├── StatusCode.java     # 状态码枚举 (20000, 20001, etc.)
│   ├── PageResult.java     # 分页结果封装
│   └── GlobalExceptionHandler.java # 全局异常处理器 (@ControllerAdvice)
├── config/                 # [配置类]
│   ├── MyBatisConfig.java  # MyBatis Plus 或 原生MyBatis配置
│   ├── WebMvcConfig.java   # 跨域(CORS)、拦截器注册
│   └── SwaggerConfig.java  # (可选) API文档配置
├── controller/             # [控制层] 接收HTTP请求，参数校验
│   ├── AnimalController.java
│   ├── TicketController.java
│   ├── MedicalController.java
│   ├── StaffController.java
│   └── LoginController.java
├── service/                # [业务逻辑层接口]
│   ├── AnimalService.java
│   ├── TicketService.java
│   ├── MedicalService.java
│   └── impl/               # [业务逻辑层实现类] 事务控制(@Transactional)在此层
│       ├── AnimalServiceImpl.java
│       ├── TicketServiceImpl.java
│       └── ...
├── mapper/                 # [数据访问层] (Dao) 与数据库交互的接口
│   ├── AnimalMapper.java
│   ├── TicketMapper.java
│   ├── MedicalMapper.java
│   └── ...
├── entity/                 # [实体类] 对应数据库表 (POJO/DO)
│   ├── Animal.java         # 对应 t_animal 表
│   ├── Species.java        # 对应 t_species 表
│   ├── MedicalRecord.java  # 对应 t_medical_record 表
│   ├── TicketOrder.java    # 对应 t_ticket_order 表
│   └── Staff.java          # 对应 t_staff 表
├── dto/                    # [数据传输对象] 用于前端表单提交接收
│   ├── AnimalDTO.java      # 接收新增/修改动物的参数
│   ├── LoginDTO.java       # 接收登录账号密码
│   └── ...
└── utils/                  # [工具类]
├── JwtUtils.java       # JWT令牌生成与解析
└── DateUtils.java      # 日期处理工具
